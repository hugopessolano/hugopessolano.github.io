Defines Pydantic models for representing API requests and their prepared data.

Includes a base `Request` model and subclasses for specific HTTP methods

for execution by `httpx` and to store/process the response.

class Request(BaseModel):
    """
    Base model for representing an API request intention.

    Contains the target endpoint configuration, HTTP method, and any
    positional arguments needed to format the final URL path.

    Attributes:
        endpoint (Endpoint): The endpoint configuration object, containing the
                             base path template and associated metadata.
        method (str): The HTTP method (e.g., "GET", "POST"). To be overridden by subclasses.
        positional_url_arguments (Optional[Tuple[str, ...]]): A tuple of string values
                                                               to substitute into the
                                                               endpoint path placeholders
                                                               (e.g., {arg_1}, {arg_2}).
        endpoint_url (str): The fully formatted URL path (without query parameters yet),
                            generated after initialization.
    """
    endpoint: Endpoint
    method: str # Must be overridden by subclasses
    positional_url_arguments: Optional[Tuple[str, ...]] = None
    # Field to store the formatted URL path after initialization
    endpoint_url: str = ""

    def model_post_init(self, __context: Any) -> None:
        """
        Post-initialization hook to format the `endpoint_url`.

        Constructs the final URL path by substituting `positional_url_arguments`
        into the placeholders (`{arg_1}`, `{arg_2}`, etc.) defined in the
        `endpoint.path`. Validates that the number of arguments provided matches
        the number of placeholders in the path template.

        Args:
            __context (Any): Pydantic validation context (unused).

        Raises:
            ValueError: If the number of provided arguments does not match the
                        number of placeholders in the endpoint path, or if
                        formatting fails for other reasons.
        """
        # Ensure any parent class post_init logic runs first (if applicable in future)
        # super().model_post_init(__context) # Not strictly needed here as BaseModel has no default

        path_template = self.endpoint.path # Use the original path from Endpoint

        if self.positional_url_arguments:
            num_provided_args = len(self.positional_url_arguments)
            # Use the count pre-calculated in the Endpoint model
            num_expected_args = self.endpoint.positional_arguments_count

            if num_provided_args != num_expected_args:
                raise ValueError(
                    f"Endpoint '{path_template}' expects {num_expected_args} positional arguments, "
                    f"but {num_provided_args} were provided: {self.positional_url_arguments}"
                )

            try:
                # Create dictionary for .format: {'arg_1': val1, 'arg_2': val2, ...}
                format_args = {
                    f"arg_{i+1}": arg
                    for i, arg in enumerate(self.positional_url_arguments)
                }
                # Perform the substitution using the original path template
                self.endpoint_url = path_template.format(**format_args)
            except (KeyError, IndexError) as e:
                # Catch potential errors if placeholders don't match arg_n format
                # or if arguments somehow mismatch during formatting.
                raise ValueError(
                    f"Error formatting endpoint path '{path_template}' with arguments "
                    f"{self.positional_url_arguments}: {e}. Ensure path uses {{arg_n}} placeholders."
                ) from e
        else:
            # If no arguments provided, check if the path expects any
            if self.endpoint.positional_arguments_count > 0:
                 raise ValueError(
                    f"Endpoint '{path_template}' expects {self.endpoint.positional_arguments_count} "
                    f"positional arguments, but none were provided."
                 )
            # No arguments needed or provided, use the path as is
            self.endpoint_url = path_template

class GetRequest(Request):
    """
    Represents a GET request, potentially including URL query parameters.

    Attributes:
        params (Optional[Dict[str, Any]]): A dictionary of query parameters to append
                                           to the URL (e.g., {"page": 1, "limit": 10}).
        method (str): Set to "GET".
    """
    params: Optional[Dict[str, Any]] = None
    method: str = "GET"



[documentos]
    def model_post_init(self, __context: Any) -> None:
        """
        Post-initialization hook for GET requests.

        Runs after the base `Request.model_post_init` to format the base URL path.
        Then, if `params` are provided, it URL-encodes them and appends them
        as a query string to the `endpoint_url`.

        Args:
            __context (Any): Pydantic validation context (unused).
        """
        super().model_post_init(__context) # Format base URL path first

        if self.params:
            # Encode the dictionary of parameters into a query string
            query_string = urlencode(self.params)
            # Append the query string to the formatted URL path
            self.endpoint_url += f"?{query_string}"

class RequestData(BaseModel):
    """
    Holds prepared data for executing a single API request via `httpx`.

    This model bridges the gap between the abstract `Request` models and the
    concrete data needed by `httpx.AsyncClient`. It also stores the response
    information after execution.

    Attributes:
        url (str): The complete, final URL for the request (including base, path, and query string).
        headers (dict): The dictionary of HTTP headers to send.
        method (str): The HTTP method (e.g., "GET", "POST").
        payload (Optional[Dict[str, Any]]): The JSON payload for POST/PUT requests.
        response_status (Optional[int]): The HTTP status code of the received response.
        response_headers (Optional[Dict[str, Any]]): The headers received in the response.
        _response (Optional[Union[Dict[str, Any], Response]]): Private attribute storing
            either the raw `httpx.Response` object initially, or the processed JSON
            dictionary after the `response` property is accessed or set.
    """
    url: str
    headers: Dict[str, str] # Changed type hint for clarity
    method: str
    payload: Optional[Dict[str, Any]] = None
    response_status: Optional[int] = None
    response_headers: Optional[Dict[str, Any]] = None
    # Private attribute to hold the raw httpx.Response or processed dict
    _response: Optional[Union[Dict[str, Any], Response]] = PrivateAttr(default=None)

    @property
    def response(self) -> Optional[Dict[str, Any]]:
        """
        Provides access to the processed response body as a dictionary.

        If the internal `_response` is still an `httpx.Response` object
        (meaning it hasn't been processed yet), this property access triggers
        the processing (`_process_assigned_response`) to extract the status code,
        headers, and JSON body. It then returns the processed dictionary body.
        If processing fails or the response was already None, returns None.

        Returns:
            Optional[Dict[str, Any]]: The JSON response body as a dictionary,
                                      or None if processing failed or no response exists.
        """
        if isinstance(self._response, Response):
            # Log a warning if accessed while still raw - indicates potential direct manipulation needed
            logging.warning(f"Accessing RequestData.response property for {self.url} while internal state is still httpx.Response. Processing now.")
            # Process the raw response into status, headers, and dict body
            self._process_assigned_response(self._response)
            # Return the processed dict, or None if processing failed
            return self._response if isinstance(self._response, dict) else None
        # If already processed or was None, return the stored value
        return self._response

    @response.setter
    def response(self, value: Optional[Union[Dict[str, Any], Response]]):
        """
        Sets the response, triggering processing if an `httpx.Response` is assigned.

        Allows assigning either a raw `httpx.Response` object (which will be
        processed) or a pre-processed dictionary/None directly.

        Args:
            value (Optional[Union[Dict[str, Any], Response]]): The response object or dictionary.
        """
        self._process_assigned_response(value)

    def _process_assigned_response(self, value: Optional[Union[Dict[str, Any], Response]]):
        """
        Internal helper to process and store the response value.

        If `value` is an `httpx.Response`, extracts status code, headers, and attempts
        to parse the JSON body. Stores these in the respective attributes and updates
        `_response` to the processed dictionary body (or an error dict if JSON parsing fails).
        If `value` is already a dict or None, stores it directly.

        Args:
            value (Optional[Union[Dict[str, Any], Response]]): The value assigned to the response.
        """
        if isinstance(value, Response):
            httpx_response: Response = value
            status_code = None
            processed_response_body = None
            headers = {}

            try:
                status_code = httpx_response.status_code
                headers = dict(httpx_response.headers) # Store headers
            except Exception as e:
                # Log error if basic attributes can't be accessed (should be rare)
                logging.error(f"Failed to get status code/headers from httpx.Response for {self.url}: {e}", exc_info=True)

            try:
                # Attempt to parse the response body as JSON
                processed_response_body = httpx_response.json()
            except json.JSONDecodeError:
                # Handle cases where the response body isn't valid JSON
                logging.warning(f"Response body is not valid JSON for URL {self.url}. Status: {status_code}")
                # Store an error dictionary instead of the raw text
                processed_response_body = {
                    "error": "JSONDecodeError",
                    "message": "Response body is not valid JSON.",
                    # Optionally include raw text, but be mindful of size/sensitivity
                    # "raw_body": httpx_response.text[:500] # Limit size
                }
            except Exception as e:
                # Handle other potential errors during .json() processing
                logging.error(f"Failed to process response body for URL {self.url}: {e}", exc_info=True)
                processed_response_body = {
                    "error": "ResponseBodyProcessingError",
                    "message": str(e)
                }

            # Update the model attributes
            self.response_status = status_code
            self.response_headers = headers
            self._response = processed_response_body # Store the processed dict (or error dict)
            logging.debug(f"Processed httpx.Response for {self.url}: Status={status_code}")

        else:
            # Value is already a dict or None
            self._response = value
            # Reset status/headers if response is explicitly set to None
            if value is None:
                 self.response_status = None
                 self.response_headers = None
            # Log if a dict is assigned directly without status (might be manual assignment)
            elif isinstance(value, dict) and self.response_status is None:
                 logging.debug(f"Dict assigned directly to response for {self.url} without explicit status code.")

    @property
    def response_text(self) -> str | None:
        """ Provides access to the raw response text if available. """
        if isinstance(self._response, Response):
             try:
                 return self._response.text
             except Exception:
                 return None # Or return an error message
        elif isinstance(self._response, dict) and "raw_body" in self._response:
             # If we stored raw_body in case of JSON error
             return self._response["raw_body"]
        # Cannot get raw text if it's already processed to dict or was None
        return None
