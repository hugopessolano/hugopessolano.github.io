"""
Defines Pydantic models for store-specific configurations, primarily language settings.

This module provides the `StoreConfig` model to manage which languages are
enabled or should be modified during operations like i18n parsing.
"""
from pydantic import BaseModel
from typing import Optional, List



[documentos]
class StoreConfig(BaseModel):
    """
    Configuration model for store language settings.

    Attributes:
        enabled_languages (List[str]): A list of language codes (e.g., "es", "en")
                                       that are considered active or enabled for the store.
                                       Used, for example, to check if an existing i18n
                                       field structure is valid.
        modify_languages (List[str]): A list of language codes for which values should
                                      be generated or modified when parsing non-i18n
                                      input into an i18n structure.
    """
    enabled_languages: List[str] = ["es", "en", "pt"]
    modify_languages: List[str] = ["es"]



#: Default store configuration instance with Spanish, English, and Portuguese enabled,
#: but modifications primarily target Spanish.
DEFAULT_CONFIG = StoreConfig()
