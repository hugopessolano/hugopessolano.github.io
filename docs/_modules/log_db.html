"""
Configures logging using Loguru, directing logs to both standard output
and an SQLite database file.

Provides a custom SQLite sink (`SQLiteSink`) to handle database writing and
configures multiple logger instances (`stdout_logger`, `db_logger`, `db_only_logger`)
with different filtering rules to control where log messages are sent based
on their intended destination (stdout only, stdout and DB, DB only).
"""
import sys, sqlite3, json, atexit
from loguru import logger
from logger_settings import STDOUT_LOGGER_SETTINGS, DB_STDOUT_LOGGER_SETTINGS, DB_LOGGER_SETTINGS, DB_ONLY_LOGGER_SETTINGS

# --- Constants ---
DB_FILE = "execution_logs.db" #: The name of the SQLite database file for storing logs.

class SQLiteSink:
    """
    A Loguru sink class that writes log records to an SQLite database.

    Handles database connection, table creation, writing formatted log records,
    and closing the connection gracefully on program exit.

    Attributes:
        db_path (str): Path to the SQLite database file.
        conn (sqlite3.Connection | None): The active SQLite connection object.
        cursor (sqlite3.Cursor | None): The database cursor for executing SQL.
    """


[documentos]
    def __init__(self, db_path: str):
        """
        Initializes the SQLiteSink.

        Establishes the database connection, creates the logs table if it
        doesn't exist, and registers the close method to be called at exit.

        Args:
            db_path (str): The file path for the SQLite database.
        """
        self.db_path: str = db_path
        self.conn: sqlite3.Connection | None = None
        self.cursor: sqlite3.Cursor | None = None
        self._connect()
        self._setup_table()
        # Ensure the database connection is closed when the program exits
        atexit.register(self.close)



    def _connect(self) -> None:
        """
        Establishes the connection to the SQLite database.

        Sets up the `conn` and `cursor` attributes. Logs an error to stderr
        if the connection fails.
        """
        try:
            # Connect to the SQLite database. check_same_thread=False is needed
            # if Loguru uses threads, though often not strictly necessary for sinks.
            self.conn = sqlite3.connect(self.db_path, check_same_thread=False)
            self.cursor = self.conn.cursor()
            # Optional: print(f"Conectado a la base de datos SQLite: {self.db_path}")
        except sqlite3.Error as e:
            # Log error directly to stderr as logger might not be fully configured yet
            print(f"Error al conectar con SQLite DB '{self.db_path}': {e}", file=sys.stderr)
            self.conn = None
            self.cursor = None

    def _setup_table(self) -> None:
        """
        Creates the 'logs' table in the database if it doesn't already exist.

        Defines the table schema including columns for log metadata and content.
        """
        if not self.cursor: return # Don't proceed if connection failed
        try:
            # SQL statement to create the table if it doesn't exist
            self.cursor.execute("""
                CREATE TABLE IF NOT EXISTS logs (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    operation_id TEXT, -- Custom field from extra data
                    timestamp TEXT NOT NULL,
                    level_name TEXT NOT NULL,
                    level_no INTEGER NOT NULL,
                    message TEXT,
                    module TEXT,
                    funcName TEXT,
                    line INTEGER,
                    extra TEXT -- Store the full 'extra' dict as JSON string
                )
            """)
            self.conn.commit() # Commit the table creation
        except sqlite3.Error as e:
            print(f"Error al crear/verificar la tabla 'logs': {e}", file=sys.stderr)

    def write(self, message: str) -> None:
        """
        Writes a log message record to the SQLite database.

        Parses the JSON formatted message from Loguru, extracts relevant fields,
        and inserts them into the 'logs' table.

        Args:
            message (str): The log message, expected to be a JSON string
                           generated by Loguru's `serialize=True` option.
        """
        if not self.conn or not self.cursor:
            # Avoids errors if connection failed during init
            print("Saltando log a SQLite: No hay conexión a la base de datos.", file=sys.stderr)
            return
        try:
            # Loguru sends the message as a JSON string when serialize=True
            record = json.loads(message)['record']

            # Prepare the data dictionary for insertion
            log_entry = {
                'operation_id': record.get('extra', {}).get('operation_id', None), # Safely get operation_id
                'timestamp': record['time']['repr'], # Use pre-formatted time string
                'level_name': record['level']['name'],
                'level_no': record['level']['no'],
                'message': record['message'],
                'module': record['module'],
                'funcName': record['function'],
                'line': record['line'],
                'extra': json.dumps(record.get('extra', {})) # Store all extra data as JSON
            }
            # Execute the INSERT statement
            self.cursor.execute("""
                INSERT INTO logs (operation_id, timestamp, level_name, level_no, message, module, funcName, line, extra)
                VALUES (:operation_id, :timestamp, :level_name, :level_no, :message, :module, :funcName, :line, :extra)
            """, log_entry)
            self.conn.commit() # Commit the transaction

        except json.JSONDecodeError as e:
            print(f"Error al decodificar el mensaje JSON del log: {e}", file=sys.stderr)
            print(f"Data recibida: {message}", file=sys.stderr)
        except sqlite3.Error as e:
            print(f"Error al escribir log en SQLite DB: {e}", file=sys.stderr)
            # Optional: Attempt rollback on error, though commit is per-write here
            # try: self.conn.rollback() except: pass
        except Exception as e:
            # Catch any other unexpected errors during writing
            print(f"Error inesperado en SQLiteSink: {e}", file=sys.stderr)

    def close(self) -> None:
        """
        Closes the connection to the SQLite database.

        Commits any pending changes and closes the connection object.
        Registered with `atexit` to run on script termination.
        """
        if self.conn:
            try:
                self.conn.commit() # Ensure final changes are saved
                self.conn.close()
                print(f"Conexión SQLite '{self.db_path}' cerrada.")
                self.conn = None
                self.cursor = None
            except sqlite3.Error as e:
                print(f"Error al cerrar la conexión SQLite: {e}", file=sys.stderr)



    def __call__(self, message: str) -> None:
        """
        Makes the SQLiteSink instance callable, allowing it to be used directly
        as a sink function in `logger.add`.

        Args:
            message (str): The log message passed by Loguru.
        """
        self.write(message)

def is_stdout_only(record: dict) -> bool:
    """
    Loguru filter function to select logs intended ONLY for stdout.

    Checks if the record's extra dictionary contains 'logger_id' == 'stdout_only'.

    Args:
        record (dict): The Loguru log record dictionary.

    Returns:
        bool: True if the record should be logged by the stdout-only sink, False otherwise.
    """
    return record["extra"].get("logger_id") == "stdout_only"

def is_db_and_stdout(record: dict) -> bool:
    """
    Loguru filter function to select logs intended for BOTH stdout and the database.

    Checks if the record's extra dictionary contains 'logger_id' == 'db_and_stdout'.

    Args:
        record (dict): The Loguru log record dictionary.

    Returns:
        bool: True if the record should be logged by the db+stdout sinks, False otherwise.
    """
    return record["extra"].get("logger_id") == "db_and_stdout"

def is_db(record: dict) -> bool:
    """
    Loguru filter function to select logs intended ONLY for the database.

    Checks if the record's extra dictionary contains 'logger_id' == 'db_only'.

    Args:
        record (dict): The Loguru log record dictionary.

    Returns:
        bool: True if the record should be logged by the db-only sink, False otherwise.
    """
    return record["extra"].get("logger_id") == "db_only"



# --- Add Sinks with Filters ---

# Sink 1: Stdout for messages marked as "stdout_only"
logger.add(
    sys.stdout,
    level=STDOUT_LOGGER_SETTINGS.level,
    format=STDOUT_LOGGER_SETTINGS.format,
    filter=is_stdout_only, # Apply the filter
    colorize=STDOUT_LOGGER_SETTINGS.colorize
)

# Sink 2: Stdout for messages marked as "db_and_stdout"
logger.add(
    sys.stdout,
    level=DB_STDOUT_LOGGER_SETTINGS.level,
    format=DB_STDOUT_LOGGER_SETTINGS.format,
    filter=is_db_and_stdout, # Apply the filter
    colorize=DB_STDOUT_LOGGER_SETTINGS.colorize
)

# Sink 3: SQLite database for messages marked as "db_and_stdout"
sqlite_sink = SQLiteSink(DB_FILE) # Create instance of the custom sink
logger.add(
    sqlite_sink, # Use the sink instance
    level=DB_LOGGER_SETTINGS.level,
    filter=is_db_and_stdout, # Apply the filter
    serialize=DB_LOGGER_SETTINGS.serialize, # MUST be True for JSON parsing in sink
    catch=DB_LOGGER_SETTINGS.catch # Loguru's exception handling
)

# Sink 4: SQLite database for messages marked as "db_only"
# Note: Using a separate sink instance might be redundant if DB_FILE is the same,
# but it's clearer for demonstrating distinct configurations.
# A single SQLiteSink instance could handle multiple logger.add calls with different filters.
sqlite_sink2 = SQLiteSink(DB_FILE)
logger.add(
    sqlite_sink2,
    level=DB_ONLY_LOGGER_SETTINGS.level,
    filter=is_db, # Apply the filter
    serialize=DB_ONLY_LOGGER_SETTINGS.serialize, # MUST be True
    catch=DB_ONLY_LOGGER_SETTINGS.catch
)

# --- Pre-bound Logger Instances ---

# Create logger instances pre-bound with the 'logger_id' for easy use
stdout_logger = logger.bind(logger_id="stdout_only") #: Logger for messages going only to stdout.
db_logger = logger.bind(logger_id="db_and_stdout") #: Logger for messages going to both stdout and the database.
db_only_logger = logger.bind(logger_id="db_only") #: Logger for messages going only to the database.


# --- Example Usage ---
if __name__ == '__main__':
    """
    Demonstrates how to use the different configured logger instances.
    This block only runs when the script is executed directly.
    """
    print("--- Iniciando prueba de loggers ---")

    # Example: Log message only to stdout
    stdout_logger.info("Este mensaje va SOLO a la consola (stdout_logger).")
    stdout_logger.debug("Este mensaje DEBUG también va SOLO a la consola.") # Will appear based on STDOUT_LOGGER_SETTINGS.level

    # Example: Log message to both stdout and database
    db_logger.info("Este mensaje va a la consola Y a la base de datos (db_logger).")
    db_logger.warning("Esta advertencia también va a ambos sitios.")
    # This debug message might only appear in stdout if DB_LOGGER_SETTINGS.level is higher (e.g., INFO)
    # but DB_STDOUT_LOGGER_SETTINGS.level is DEBUG.
    db_logger.debug("Este mensaje DEBUG de db_logger NO aparecerá en la BD (nivel INFO), pero SÍ en consola (porque el sink de stdout para db_logger lo permite).")

    # Example: Log message only to database
    db_only_logger.error("Este error va SOLO a la base de datos (db_only_logger).")

    # Example: Binding additional context dynamically
    stdout_logger.bind(user="Hugo").info("Acción específica de Hugo (solo consola).")
    db_logger.bind(request_id="xyz789", operation_id="op123").error("Error en request (consola y BD).") # operation_id will be in DB

    print("--- Prueba de loggers finalizada ---")
